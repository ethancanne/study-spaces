const Mongoose = require("mongoose");
const Schema = Mongoose.Schema;

/**
 * @author Cliff Croom
 * @date   01/11/2022
 */
const PostSchema = new Schema({
    attachment: {
        type: String,
        required: false
    },
    creator: {
        type: String,
        required: true
    },
    message: {
        type: String,
        required: true
    },
    responses: {
        type: [Object],
        required: true
    },
    timeStamp: {
        type: Date,
        required: true
    },
    title: {
        type: String,
        required: true
    },
    type: {
        type: String,
        required: true
    }
});
PostSchema.set("toObject", {
    versionKey: false,
    transform: (document, object) => {
        delete object.__v;
        return object;
    }
});
const PostCollectionName = Configuration.getPostCollectionName();
const PostModel = Mongoose.model(PostCollectionName, PostSchema);
